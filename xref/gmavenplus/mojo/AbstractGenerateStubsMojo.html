121     protected Set<File> getSources() {

202     /**
203      * @param sources
204      * @param outputDirectory
205      * @throws ClassNotFoundException
206      * @throws InvocationTargetException
207      * @throws IllegalAccessException
208      * @throws InstantiationException
209      */
210     protected void doStubGeneration(Set<File> sources, File outputDirectory) throws ClassNotFoundException, InvocationTargetException, IllegalAccessException, InstantiationException {
211         // get classes we need with reflection
212         Class compilerConfigurationClass = Class.forName("org.codehaus.groovy.control.CompilerConfiguration");
213         Class javaStubCompilationUnitClass = Class.forName("org.codehaus.groovy.tools.javac.JavaStubCompilationUnit");
214         Class groovyClassLoaderClass = Class.forName("groovy.lang.GroovyClassLoader");
215 
216         // set up compile options
217         Object compilerConfiguration = ReflectionUtils.findConstructor(compilerConfigurationClass).newInstance();
218         ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "setDebug", boolean.class), compilerConfiguration, debug);
219         ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "setVerbose", boolean.class), compilerConfiguration, verbose);
220         ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "setWarningLevel", int.class), compilerConfiguration, warningLevel);
221         ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "setTolerance", int.class), compilerConfiguration, tolerance);
222         if (sourceEncoding != null) {
223             ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "setSourceEncoding", String.class), compilerConfiguration, sourceEncoding);
224         }
225         ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "setTargetDirectory", String.class), compilerConfiguration, outputDirectory.getAbsolutePath());
226         ClassLoader parent = ClassLoader.getSystemClassLoader();
227         Object groovyClassLoader = ReflectionUtils.findConstructor(groovyClassLoaderClass, ClassLoader.class, compilerConfigurationClass).newInstance(parent, compilerConfiguration);
228         Object javaStubCompilationUnit = ReflectionUtils.findConstructor(javaStubCompilationUnitClass, compilerConfigurationClass, groovyClassLoaderClass, File.class).newInstance(compilerConfiguration, groovyClassLoader, outputDirectory);
229         getLog().debug("Compiling " + sources.size() + " sources");
230         for (File source : sources) {
231             URL url = null;
232             try {
233                 url = source.toURI().toURL();
234             } catch (MalformedURLException e) {
235                 getLog().error("Unable to add source file " + source.getAbsolutePath() + " for stub generation", e);
236             }
237             ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(javaStubCompilationUnitClass, "addSource", URL.class), javaStubCompilationUnit, url);
238         }
239 
240         // compile the classes
241         ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(javaStubCompilationUnitClass, "compile"), javaStubCompilationUnit);
242 
243         // log compiled classes
244         Integer stubCount = (Integer) ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(javaStubCompilationUnitClass, "getStubCount"), javaStubCompilationUnit);
245         if (getLog().isDebugEnabled()) {
246             getLog().debug("Generated " + stubCount + " stubs: ");
247         }
248     }

250     /**
251      * This is a fix for fix for http://jira.codehaus.org/browse/MGROOVY-187
252      * It modifies the dates of the created stubs to 1970, ensuring that the Java
253      * compiler will not come along and overwrite perfectly good compiled Groovy
254      * just because it has a newer source stub.  Basically, this prevents the
255      * stubs from causing a side effect with the Java compiler, but still allows
256      * the stubs to work with JavaDoc.  Ideally, the code for this should be
257      * added to the code that creates the stubs.
258      *
259      * @param stubs the files on which to reset the modified date
260      */
261     protected void resetStubModifiedDates(Set<File> stubs ) {
262         for (File file : stubs) {
263             file.setLastModified(0L);
264         }
265     }
