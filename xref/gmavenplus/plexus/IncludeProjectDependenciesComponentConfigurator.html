25  /**
26   * A custom ComponentConfigurator which adds the project's compile time classpath elements
27   *
28   * @author Brian Jackson
29   * @since Aug 1, 2008 3:04:17 PM
30   *
31   * @plexus.component role="org.codehaus.plexus.component.configurator.ComponentConfigurator"
32   *                   role-hint="include-project-dependencies"
33   * @plexus.requirement role="org.codehaus.plexus.component.configurator.converters.lookup.ConverterLookup"
34   *                     role-hint="default"
35   */
36  public class IncludeProjectDependenciesComponentConfigurator extends AbstractComponentConfigurator {
37      // TODO: add logging?
38  //    private static final Logger LOGGER = LoggerFactory.getLogger(IncludeProjectDependenciesComponentConfigurator.class);
39  
40      public void configureComponent(Object component, PlexusConfiguration configuration,
41                                     ExpressionEvaluator expressionEvaluator, ClassRealm containerRealm,
42                                     ConfigurationListener listener) throws ComponentConfigurationException {
43          addProjectDependenciesToClassRealm(expressionEvaluator, containerRealm);
44          converterLookup.registerConverter(new ClassRealmConverter(containerRealm));
45          ObjectWithFieldsConverter converter = new ObjectWithFieldsConverter();
46          converter.processConfiguration(converterLookup, component, containerRealm.getClassLoader(), configuration, expressionEvaluator, listener);
47      }
48  
49      private void addProjectDependenciesToClassRealm(ExpressionEvaluator expressionEvaluator, ClassRealm containerRealm) throws ComponentConfigurationException {
50          List compileClasspathElements;
51  
52          try {
53              compileClasspathElements = (List) expressionEvaluator.evaluate("${project.compileClasspathElements}");
54          } catch (ExpressionEvaluationException e) {
55              throw new ComponentConfigurationException("There was a problem evaluating: ${project.compileClasspathElements}.", e);
56          }
57  
58          // Add the project dependencies to the ClassRealm
59          final URL[] urls = buildURLs(compileClasspathElements);
60          for (URL url : urls) {
61              containerRealm.addConstituent(url);
62          }
63      }
64  
65      private URL[] buildURLs(List compileClasspathElements) throws ComponentConfigurationException {
66          List<URL> urls = new ArrayList<URL>(compileClasspathElements.size());
67          for (Object element : compileClasspathElements) {
68              try {
69                  final URL url = new File((String) element).toURI().toURL();
70                  urls.add(url);
71  //                LOGGER.debug("Added to project class loader: {}", url);
72              } catch (MalformedURLException e) {
73                  throw new ComponentConfigurationException("Unable to access project dependency: " + element + ".", e);
74              }
75          }
76  
77          return urls.toArray(new URL[urls.size()]);
78      }
