111     /**
112      * Logs the version of groovy used by this mojo.
113      *
114      * @param goal The goal to mention in the log statement showing Groovy version
115      */
116     protected void logGroovyVersion(final String goal) {
117         if (getLog().isInfoEnabled()) {
118             String logMessage = "Using Groovy " + getGroovyVersion();
119             if (isGroovyIndy()) {
120                 logMessage += "-indy";
121             }
122             getLog().info(logMessage + " from project compile classpath to perform " + goal + ".");
123         }
124     }
125 
126     /**
127      * Gets the version string of Groovy used from the dependency information.
128      *
129      * @return The version string of Groovy used by the project
130      */
131     protected String getGroovyVersionString() {
132         String groovyVersion = null;
133 
134         /*
135          * You can call InvokerHelper.getVersion() for versions 1.0 - 1.8.x but
136          * not for 1.9+.
137          * You can call GroovySystem.getVersion() for versions 1.6.6+.
138          * And for some reason InvokerHelper.getVersion() was returning an empty
139          * String for 1.5.0, so I decided to just get it from the dependency itself.
140          */
141         Artifact groovyDependency = getGroovyDependency();
142 
143         if (groovyDependency == null) {
144             getLog().error("Unable to determine Groovy version.  Is Groovy declared as a dependency?");
145         } else {
146             groovyVersion = groovyDependency.getVersion();
147         }
148 
149         return groovyVersion;
150     }
151 
152     /**
153      * Gets the version of Groovy used from the dependency information.
154      *
155      * @return The version of Groovy used by the project
156      */
157     protected Version getGroovyVersion() {
158         try {
159             return Version.parseFromString(getGroovyVersionString());
160         } catch (Exception e) {
161             getLog().error("Unable to determine Groovy version.  Is Groovy declared as a dependency?");
162             return null;
163         }
164     }
165 
166     /**
167      * Gets the version of Groovy used from the dependency information.
168      *
169      * @return <code>true</code> if the version of Groovy uses InvokeDynamic,
170      *         <code>false</code> if not or Groovy dependency cannot be found.
171      */
172     protected boolean isGroovyIndy() {
173         boolean isGroovyIndy = false;
174 
175         Artifact groovyDependency = getGroovyDependency();
176         if (groovyDependency == null) {
177             getLog().error("Unable to determine Groovy version.  Is Groovy declared as a dependency?");
178         } else if ("indy".equals(groovyDependency.getClassifier())) {
179             isGroovyIndy = true;
180         }
181 
182         return isGroovyIndy;
183     }
184 
185     /**
186      * Gets the Groovy dependency used by the project.
187      *
188      * @return The Groovy dependency used by the project
189      */
190     protected Artifact getGroovyDependency() {
191         Artifact groovyDependency = null;
192 
193         if (pluginArtifacts != null) {
194             for (Object art : pluginArtifacts) {
195                 Artifact artifact = (Artifact) art;
196                 if (isGroovyGroupId(artifact) && isGroovyArtifactId(artifact) && artifact.getType().equals("jar")) {
197                     groovyDependency = artifact;
198                     break;
199                 }
200             }
201         }
202 
203         if (groovyDependency == null && project.getCompileDependencies() != null) {
204             for (Object dep : project.getCompileDependencies()) {
205                 Dependency dependency = (Dependency) dep;
206                 if (isGroovyGroupId(dependency) && isGroovyArtifactId(dependency) && dependency.getType().equals("jar")) {
207                     groovyDependency = new DefaultArtifact(dependency.getGroupId(), dependency.getArtifactId(), VersionRange.createFromVersion(dependency.getVersion()), dependency.getScope(), dependency.getType(), dependency.getClassifier() != null ? dependency.getClassifier() : "", null);
208                     break;
209                 }
210             }
211         }
212 
213         if (groovyDependency == null && project.getTestDependencies() != null) {
214             for (Object dep : project.getTestDependencies()) {
215                 Dependency dependency = (Dependency) dep;
216                 if (isGroovyGroupId(dependency) && isGroovyArtifactId(dependency) && dependency.getType().equals("jar")) {
217                     groovyDependency = new DefaultArtifact(dependency.getGroupId(), dependency.getArtifactId(), VersionRange.createFromVersion(dependency.getVersion()), dependency.getScope(), dependency.getType(), dependency.getClassifier() != null ? dependency.getClassifier() : "", null);
218                     break;
219                 }
220             }
221         }
222 
223         return groovyDependency;
224     }

226     /**
227      * Whether the groupId of the dependency is Groovy's groupId.
228      *
229      * @param dependency The dependency to inspect
230      * @return <code>true</code> if the dependency's groupId is a Groovy groupId, <code>false</code> otherwise
231      */
232     protected boolean isGroovyGroupId(final Dependency dependency) {
233         return dependency.getGroupId().equals("org.codehaus.groovy") || dependency.getGroupId().equals("groovy");
234     }
235 
236     /**
237      * Whether the artifactId of the dependency is Groovy's artifactId.
238      *
239      * @param dependency The dependency to inspect
240      * @return <code>true</code> if the dependency's groupId is a Groovy groupId, <code>false</code> otherwise
241      */
242     protected boolean isGroovyArtifactId(final Dependency dependency) {
243         return dependency.getArtifactId().equals("groovy-all") || dependency.getArtifactId().equals("groovy-all-minimal")
244                                     || dependency.getArtifactId().equals("groovy") || dependency.getArtifactId().equals("groovy-all-jdk14")
245                                     || dependency.getArtifactId().equals("groovy-jdk14");
246     }
247 
248 
249     /**
250      * Whether the groupId of the dependency is Groovy's groupId.
251      *
252      * @param artifact The dependency to inspect
253      * @return <code>true</code> if the dependency's groupId is a Groovy groupId, <code>false</code> otherwise
254      */
255     protected boolean isGroovyGroupId(final Artifact artifact) {
256         return artifact.getGroupId().equals("org.codehaus.groovy") || artifact.getGroupId().equals("groovy");
257     }
258 
259     /**
260      * Whether the artifactId of the dependency is Groovy's artifactId.
261      *
262      * @param artifact The dependency to inspect
263      * @return <code>true</code> if the dependency's groupId is a Groovy groupId, <code>false</code> otherwise
264      */
265     protected boolean isGroovyArtifactId(final Artifact artifact) {
266         return artifact.getArtifactId().equals("groovy-all") || artifact.getArtifactId().equals("groovy-all-minimal")
267                 || artifact.getArtifactId().equals("groovy") || artifact.getArtifactId().equals("groovy-all-jdk14")
268                 || artifact.getArtifactId().equals("groovy-jdk14");
269     }
270 
271     /**
272      * Determines whether the version of Java executing this mojo supports invokedynamic (is at least 1.7).
273      *
274      * @return <code>true</code> if the running Java supports invokedynamic, <code>false</code> otherwise
275      */
276     protected boolean isJavaSupportIndy() {
277         return getJavaVersion().compareTo(new Version(1, 7), false) >= 0;
278     }
279 
280     /**
281      * Gets the version of Java executing this mojo as a Version object.
282      *
283      * @return a Version object of the running Java version
284      */
285     protected Version getJavaVersion() {
286         return Version.parseFromString(getJavaVersionString());
287     }
288 
289     /**
290      * Gets the version of Java executing this mojo as a String.
291      *
292      * @return a String of the running Java version
293      */
294     protected String getJavaVersionString() {
295         return System.getProperty("java.version");
296     }
297 
298     /**
299      * Determines whether this mojo can be run with the version of Groovy supplied.
300      *
301      * @return <code>true</code> only if the version of Groovy supports this mojo.
302      */
303     protected boolean groovyVersionSupportsAction() {
304         return getGroovyVersion() != null && getGroovyVersion().compareTo(minGroovyVersion) >= 0;
305     }
