17  package org.codehaus.gmavenplus.mojo;
18  
19  import java.lang.reflect.InvocationTargetException;
20  import org.apache.maven.plugin.MojoExecutionException;
21  import org.apache.maven.plugin.MojoFailureException;
22  
23  
24  /**
25   * Generates stubs for the main sources.
26   * Note that this mojo requires Groovy >= 1.7.0.
27   *
28   * @author Keegan Witt
29   *
30   * @goal generateStubs
31   * @phase generate-sources
32   * @threadSafe
33   */
34  public class GenerateStubsMojo extends AbstractGenerateStubsMojo {
35  
36      /**
37       * Executes this mojo.
38       *
39       * @throws MojoExecutionException If an unexpected problem occurs. Throwing this exception causes a "BUILD ERROR" message to be displayed
40       * @throws MojoFailureException If an expected problem (such as a compilation failure) occurs. Throwing this exception causes a "BUILD FAILURE" message to be displayed
41       */
42      public void execute() throws MojoExecutionException, MojoFailureException {
43          if (groovyVersionSupportsAction()) {
44              logGroovyVersion("generateStubs");
45  
46              try {
47                  doStubGeneration(getSources(), stubsOutputDirectory);
48                  resetStubModifiedDates(getStubs());
49  
50                  // add stubs to project source so the Maven Compiler Plugin can find them
51                  project.addCompileSourceRoot(stubsOutputDirectory.getAbsolutePath());
52  
53                  // log generated stubs
54                  int stubCount = getStubs().size();
55                  getLog().info("Generated " + stubCount + " stubs.");
56              } catch (ClassNotFoundException e) {
57                  throw new MojoExecutionException("Unable to get a Groovy class from classpath.  Do you have Groovy as a compile dependency in your project?", e);
58              } catch (InvocationTargetException e) {
59                  throw new MojoExecutionException("Error occurred while calling a method on a Groovy class from classpath.", e);
60              } catch (InstantiationException e) {
61                  throw new MojoExecutionException("Error occurred while instantiating a Groovy class from classpath.", e);
62              } catch (IllegalAccessException e) {
63                  throw new MojoExecutionException("Unable to access a method on a Groovy class from classpath.", e);
64              }
65          } else {
66              getLog().error("Your Groovy version (" + getGroovyVersion() + ") doesn't support stub generation.  The minimum version of Groovy required is " + MIN_GROOVY_VERSION + ".  Skipping stub generation.");
67          }
68      }
69  
70  }
