17  package org.codehaus.gmavenplus.util;
18  
19  import com.google.common.io.Files;
20  
21  import java.io.File;
22  import java.net.URI;
23  import java.util.HashSet;
24  import java.util.Set;
25  
26  
27  /**
28   * This class exists solely to trick
29   * <a href="http://groovy.codehaus.org/api/org/codehaus/groovy/tools/javac/JavaStubCompilationUnit.html#addSource%28java.io.File%29">JavaStubCompilationUnit.addSource(java.io.File)</a>
30   * into letting us use files that don't end in ".groovy" (as a workaround for <a href="http://jira.codehaus.org/browse/GROOVY-5021">GROOVY-5021</a>).
31   *
32   * @author Keegan Witt
33   */
34  public class DotGroovyFile extends File {
35      private Set<String> scriptExtensions = new HashSet<String>();
36  
37      /**
38       * Constructs a new DotGroovyFile object with the specified parameters.
39       *
40       * @param pathname Pathname to use to create DotGroovyFile
41       */
42      public DotGroovyFile(String pathname) {
43          super(pathname);
44      }
45  
46      /**
47       * Constructs a new DotGroovyFile object with the specified parameters.
48       *
49       * @param parent Parent pathname to use to create DotGroovyFile
50       * @param child Child pathname to use to create DotGroovyFile
51       */
52      public DotGroovyFile(String parent, String child) {
53          super(parent, child);
54      }
55  
56      /**
57       * Constructs a new DotGroovyFile object with the specified parameters.
58       *
59       * @param parent Parent file to use to create DotGroovyFile
60       * @param child Child pathname to use to create DotGroovyFile
61       */
62      public DotGroovyFile(File parent, String child) {
63          super(parent, child);
64      }
65  
66      /**
67       * Constructs a new DotGroovyFile object with the specified parameters.
68       *
69       * @param uri URI to use to create DotGroovyFile
70       */
71      public DotGroovyFile(URI uri) {
72          super(uri);
73      }
74  
75      /**
76       * A convenience constructor to turn a regular file into a DotGroovyFile.
77       *
78       * @param file File to use to create DotGroovyFile
79       */
80      public DotGroovyFile(File file) {
81          this(file.getAbsolutePath());
82      }
83  
84      /**
85       * Default Groovy file extensions (currently '.groovy', '.gvy', '.gy', and '.gsh').
86       *
87       * @return The default Groovy file extensions
88       */
89      public static Set<String> defaultScriptExtensions() {
90          Set<String> defaultScriptExtensions = new HashSet<String>();
91  
92          defaultScriptExtensions.add("groovy");
93          defaultScriptExtensions.add("gvy");
94          defaultScriptExtensions.add("gy");
95          defaultScriptExtensions.add("gsh");
96  
97          return defaultScriptExtensions;
98      }
99  
100     /**
101      * A method to lie about the file extension and say it is ".groovy".
102      *
103      * @return Filename with forced <tt>.groovy</tt> extension
104      */
105     public String getName() {
106         if (scriptExtensions != null && !scriptExtensions.isEmpty() && scriptExtensions.contains(Files.getFileExtension(super.getAbsolutePath()))) {
107             return Files.getNameWithoutExtension(super.getName()) + ".groovy";
108         } else {
109             return super.getName();
110         }
111     }
112 
113     /**
114      * Gets the script extensions for this Groovy file.
115      *
116      * @return The script extensions for this Groovy file
117      */
118     public Set<String> getScriptExtensions() {
119         return scriptExtensions;
120     }
121 
122     /**
123      * Sets the script extensions for this Groovy file.
124      *
125      * @param newScriptExtensions The script extensions to set on this Groovy file
126      * @return This object (for fluent invocation)
127      */
128     public DotGroovyFile setScriptExtensions(Set<String> newScriptExtensions) {
129         this.scriptExtensions = newScriptExtensions;
130         return this;
131     }

133 }
